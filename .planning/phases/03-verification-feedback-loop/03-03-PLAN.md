       ```go
       func runVerifyWriteResults(out io.Writer, specsDir string, ws *state.WorkflowState, reportPath string) error {
           // 1. Read and parse report JSON
           data, err := os.ReadFile(reportPath)
           report, err := verifier.ParseVerifierReport(data)

           // 2. Write verification.md (D-01)
           changeDir := filepath.Join(specsDir, "changes", ws.ChangeName)
           verifier.WriteVerificationReport(changeDir, report)

           // 3. Write gap-report.md if failures (D-05)
           verifier.WriteGapReport(changeDir, report)

           // 4. Update verification-status.json (D-04) — MUST items only
           vs := spec.VerificationStatus{
               ChangeName: ws.ChangeName,
               VerifiedAt: time.Now(),
               Requirements: make(map[string]spec.ItemStatus),
           }
           for _, r := range report.Results {
               if r.Keyword == "MUST" {
                   if r.Pass { vs.Requirements[r.ID] = spec.StatusDone }
                   else { vs.Requirements[r.ID] = spec.StatusBlocked }
               }
           }
           spec.WriteVerificationStatus(changeDir, vs)

           // 5. State transition (D-16)
           pass := report.MustPass
           ws.VerifyPass = &pass
           if report.MustPass {
               state.Transition(ws, state.PhaseVerified)
           }
           state.SaveState(specsDir, *ws)

           // 6. Terminal summary (D-01 — lipgloss styled)
           // Print summary: "MUST: X/Y passed", "SHOULD: ...", "MAY: ..."
           return nil
       }
       ```

    2. Implement `runArchive`:
       ```go
       func runArchive(cmd *cobra.Command, args []string) error {
           specsDir, err := spec.DetectSpecDir(".")
           ws, err := state.LoadState(specsDir)

           // Gate 1: state must be verified (D-17)
           if ws.Phase != state.PhaseVerified {
               return fmt.Errorf("cannot archive: phase is %s, must be verified", ws.Phase)
           }

           // Gate 2: all MUST items must be DONE (D-17)
           changeDir := filepath.Join(specsDir, "changes", ws.ChangeName)
           vs, err := spec.ReadVerificationStatus(changeDir)
           change, err := spec.ParseChange(changeDir)
           for _, r := range change.Specs {
               if r.Keyword == spec.Must {
                   id := verifier.StableID(r)
                   status, ok := vs.Requirements[id]
                   if !ok || status != spec.StatusDone {
                       return fmt.Errorf("cannot archive: MUST item %q is not done", truncate(r.Text, 60))
                   }
               }
           }

           // UAT prompt (D-09) — prompt but don't block
           skipPrompt, _ := cmd.Flags().GetBool("yes")
           if !skipPrompt && isInteractive() {
               fmt.Fprint(cmd.OutOrStdout(), "Run UAT first? [y/N] ")
               // Read response; if "y", print hint: "Run /mysd:uat to start UAT. Continuing with archive..."
               // Regardless of answer, continue
           }

           // Archive: move directory (D-08)
           archiveDir := filepath.Join(specsDir, "archive", ws.ChangeName)
           if err := os.MkdirAll(filepath.Dir(archiveDir), 0755); err != nil { return err }
           // Save state snapshot to archive dir before moving (Pitfall 5)
           if err := state.SaveState(changeDir, ws); err != nil { /* log warning, don't fail */ }
           // Rename snapshot to ARCHIVED-STATE.json
           archivedStateSrc := filepath.Join(changeDir, "STATE.json")
           archivedStateDst := filepath.Join(changeDir, "ARCHIVED-STATE.json")
           os.Rename(archivedStateSrc, archivedStateDst) // best-effort

           if err := moveDir(changeDir, archiveDir); err != nil { return err }

           // Transition state (D-08)
           state.Transition(&ws, state.PhaseArchived)
           state.SaveState(specsDir, ws)

           fmt.Fprintf(cmd.OutOrStdout(), "Archived %s to %s\n", ws.ChangeName, archiveDir)
           return nil
       }
       ```
