    ```go
    package cmd

    import (
        "fmt"
        "path/filepath"
        "strings"

        "github.com/spf13/cobra"
        "github.com/spf13/viper"
        "github.com/xenciscbc/mysd/internal/config"
        "github.com/xenciscbc/mysd/internal/output"
    )

    // Fixed role order for deterministic output (RESEARCH.md Open Question 1)
    var knownRoles = []string{
        "spec-writer", "designer", "planner", "executor", "verifier",
        "fast-forward", "researcher", "advisor", "proposal-writer", "plan-checker",
    }

    var modelCmd = &cobra.Command{
        Use:   "model",
        Short: "Display or set model profile",
        RunE:  runModelRead,
    }

    var modelSetCmd = &cobra.Command{
        Use:   "set <profile>",
        Short: "Set model profile (quality, balanced, budget)",
        Args:  cobra.ExactArgs(1),
        RunE:  runModelSet,
    }

    func init() {
        modelCmd.AddCommand(modelSetCmd)
        rootCmd.AddCommand(modelCmd)
    }
    ```

    `runModelRead(cmd *cobra.Command, args []string) error`:
    1. Load config via `config.Load(".")`
    2. Get profile from `cfg.ModelProfile` (default "balanced")
    3. Create `output.NewPrinter(cmd.OutOrStdout())`
    4. Print header line: `fmt.Fprintf(w, "Profile: %s\n\n", profile)`
    5. Print table header: `fmt.Fprintf(w, "%-20s %s\n", "Role", "Model")`
    6. Print separator: `fmt.Fprintf(w, "%s\n", strings.Repeat("-", 40))`
    7. For each role in `knownRoles`: resolve model via `config.ResolveModel(role, profile, cfg.ModelOverrides)`, print `fmt.Fprintf(w, "%-20s %s\n", role, model)`

    NOTE: Use `cmd.OutOrStdout()` directly with fmt.Fprintf for non-TTY-safe plain text output. Do NOT use lipgloss table rendering in this plan (TTY lipgloss table is enhancement, plain text satisfies D-11 requirement). The output format is: Profile header + Role/Model two-column table.

    Create `cmd/model_test.go` with tests:
    - Execute model command via cobra's `cmd.SetOut(buf)` + `cmd.Execute()` pattern
    - Use `bytes.Buffer` as output capture
    - Assert output contains "Profile: balanced", all 10 role names, "claude-sonnet-4-5"

    3. Write to config using Viper (per RESEARCH.md Pattern 3):
       ```go
       v := viper.New()
       configPath := filepath.Join(".", ".claude", "mysd.yaml")
       v.SetConfigFile(configPath)
       // CRITICAL (Pitfall 1): Must ReadInConfig first to preserve other fields
       if err := v.ReadInConfig(); err != nil {
           // If file doesn't exist, that's OK — SafeWriteConfig will create it
       }
       v.Set("model_profile", profile)
       if err := v.WriteConfig(); err != nil {
           // Try SafeWriteConfig if file doesn't exist yet
           if err := v.SafeWriteConfig(); err != nil {
               return fmt.Errorf("write config: %w", err)
           }
       }
       ```
    4. Print success: `p.Success(fmt.Sprintf("Model profile set to: %s", profile))`
