    type ScanContext struct {
        RootDir       string        `json:"root_dir"`

        ExistingSpecs []string      `json:"existing_specs"`
        ExcludedDirs  []string      `json:"excluded_dirs"`

    Implementation details for BuildScanContext:
    - Use `filepath.WalkDir` (stdlib) to traverse `root`
    - Skip directories whose name starts with "." (hidden dirs like .git, .specs)
    - Skip directories in the `exclude` list (case-sensitive name match)
    - For each `.go` file found: determine its package directory via `filepath.Dir`
    - Separate `_test.go` files into TestFiles, regular `.go` files into GoFiles
    - Group files by directory into PackageInfo structs
    - Set PackageInfo.Name to the relative directory path from root (use forward slashes via `filepath.ToSlash`)
    - Detect existing specs: after WalkDir, use `spec.DetectSpecDir(root)` to find the specs directory. For each package, check if `filepath.Join(specsDir, "changes", pkgName)` exists — if so, set HasSpec=true and add to ExistingSpecs
    - If DetectSpecDir returns error (no specs dir), all HasSpec=false and ExistingSpecs=nil — this is fine for first-time scan
    - TotalFiles = sum of all GoFiles + TestFiles across all packages
    - ExcludedDirs = the exclude slice passed in
    - RootDir = root parameter

    **Part A: cmd/scan.go** — Create scan command following verify.go pattern exactly (per D-01):

    ```go
    package cmd

    import (
        "encoding/json"
        "fmt"
        "io"

        "github.com/mysd/internal/scanner"
        "github.com/spf13/cobra"
    )

    var scanCmd = &cobra.Command{
        Use:   "scan",
        Short: "Scan codebase and output metadata for spec generation",
        RunE:  runScan,
    }

    func init() {
        scanCmd.Flags().Bool("context-only", false, "Output scan context as JSON (for /mysd:scan agent consumption)")
        scanCmd.Flags().StringSlice("exclude", nil, "Directories to exclude from scan")
        rootCmd.AddCommand(scanCmd)
    }

    func runScan(cmd *cobra.Command, args []string) error {
        contextOnly, _ := cmd.Flags().GetBool("context-only")
        if !contextOnly {
            return fmt.Errorf("usage: mysd scan --context-only [--exclude dir1,dir2]\nDirect execution via /mysd:scan")
        }
        exclude, _ := cmd.Flags().GetStringSlice("exclude")
        return runScanContextOnly(cmd.OutOrStdout(), ".", exclude)
    }

    func runScanContextOnly(out io.Writer, root string, exclude []string) error {
        ctx, err := scanner.BuildScanContext(root, exclude)
        if err != nil {
            return fmt.Errorf("build scan context: %w", err)
        }
        data, err := json.MarshalIndent(ctx, "", "  ")
        if err != nil {
            return fmt.Errorf("marshal scan context: %w", err)
        }
        _, err = out.Write(data)
        return err
    }
    ```
